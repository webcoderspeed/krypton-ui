# Breadcrumb

Displays the path to the current resource using a hierarchy of links. Helps users understand their current location within a website or application.

## Installation

```bash
npm install @radix-ui/react-slot
npm install lucide-react
```

## Import

```tsx
import {
  Breadcrumb,
  BreadcrumbEllipsis,
  BreadcrumbItem,
  BreadcrumbLink,
  BreadcrumbList,
  BreadcrumbPage,
  BreadcrumbSeparator,
} from "@/components/ui/breadcrumb"
```

## API Reference

### Breadcrumb

The root container for the breadcrumb navigation.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `separator` | `ReactNode` | - | Custom separator element |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The breadcrumb content |

### BreadcrumbList

The list container for breadcrumb items.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The breadcrumb items |

**Default Styles:**
- `flex flex-wrap items-center` - Flexible layout
- `gap-1.5 break-words` - Spacing and text wrapping
- `text-sm text-muted-foreground` - Typography
- `sm:gap-2.5` - Responsive spacing

### BreadcrumbItem

A single item in the breadcrumb navigation.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The item content |

**Default Styles:**
- `inline-flex items-center gap-1.5` - Inline flex layout

### BreadcrumbLink

A clickable link within a breadcrumb item.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | `false` | Render as child component (for custom link components) |
| `className` | `string` | - | Additional CSS classes |
| `href` | `string` | - | The link destination |
| `children` | `ReactNode` | - | The link content |

**Default Styles:**
- `transition-colors hover:text-foreground` - Hover effects

### BreadcrumbPage

Represents the current page in the breadcrumb (non-clickable).

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The page content |

**Default Styles:**
- `font-normal text-foreground` - Typography styling
- `role="link" aria-disabled="true" aria-current="page"` - Accessibility attributes

### BreadcrumbSeparator

The separator between breadcrumb items.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | `<ChevronRight />` | Custom separator content |
| `className` | `string` | - | Additional CSS classes |

**Default Styles:**
- `[&>svg]:size-3.5` - Icon sizing
- `role="presentation" aria-hidden="true"` - Accessibility attributes

### BreadcrumbEllipsis

Indicates collapsed/hidden breadcrumb items.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |

**Default Styles:**
- `flex h-9 w-9 items-center justify-center` - Centered layout
- `role="presentation" aria-hidden="true"` - Accessibility attributes

## Features

- **Accessible**: Proper ARIA attributes and semantic HTML
- **Flexible**: Support for custom separators and links
- **Responsive**: Built-in responsive behavior
- **Customizable**: Easy to style and extend
- **Framework Agnostic**: Works with any routing solution
- **Ellipsis Support**: Handle long breadcrumb chains gracefully

## Accessibility

- Uses semantic `nav` element with `aria-label="breadcrumb"`
- Proper `role` and `aria-current` attributes
- Screen reader friendly with hidden decorative elements
- Keyboard navigation support
- Sufficient color contrast for all states

2. **Spacing**: Adjust gap and padding classes
3. **Colors**: Change text and hover colors
4. **Separators**: Use custom icons or text

## Common Patterns

### Dynamic Breadcrumbs

```tsx
interface BreadcrumbItem {
  label: string
  href?: string
}

interface DynamicBreadcrumbProps {
  items: BreadcrumbItem[]
}

function DynamicBreadcrumb({ items }: DynamicBreadcrumbProps) {
  return (
    <Breadcrumb>
      <BreadcrumbList>
        {items.map((item, index) => (
          <React.Fragment key={index}>
            <BreadcrumbItem>
              {item.href ? (
                <BreadcrumbLink href={item.href}>{item.label}</BreadcrumbLink>
              ) : (
                <BreadcrumbPage>{item.label}</BreadcrumbPage>
              )}
            </BreadcrumbItem>
            {index < items.length - 1 && <BreadcrumbSeparator />}
          </React.Fragment>
        ))}
      </BreadcrumbList>
    </Breadcrumb>
  )
}
```

### E-commerce Breadcrumb

```tsx
function ProductBreadcrumb({ category, subcategory, product }) {
  return (
    <Breadcrumb>
      <BreadcrumbList>
        <BreadcrumbItem>
          <BreadcrumbLink href="/">Home</BreadcrumbLink>
        </BreadcrumbItem>
        <BreadcrumbSeparator />
        <BreadcrumbItem>
          <BreadcrumbLink href="/shop">Shop</BreadcrumbLink>
        </BreadcrumbItem>
        <BreadcrumbSeparator />
        <BreadcrumbItem>
          <BreadcrumbLink href={`/shop/${category.slug}`}>
            {category.name}
          </BreadcrumbLink>
        </BreadcrumbItem>
        {subcategory && (
          <>
            <BreadcrumbSeparator />
            <BreadcrumbItem>
              <BreadcrumbLink href={`/shop/${category.slug}/${subcategory.slug}`}>
                {subcategory.name}
              </BreadcrumbLink>
            </BreadcrumbItem>
          </>
        )}
        <BreadcrumbSeparator />
        <BreadcrumbItem>
          <BreadcrumbPage>{product.name}</BreadcrumbPage>
        </BreadcrumbItem>
      </BreadcrumbList>
    </Breadcrumb>
  )
}
```

### Documentation Breadcrumb

```tsx
function DocsBreadcrumb({ path }) {
  const segments = path.split('/').filter(Boolean)
  
  return (
    <Breadcrumb>
      <BreadcrumbList>
        <BreadcrumbItem>
          <BreadcrumbLink href="/">Home</BreadcrumbLink>
        </BreadcrumbItem>
        {segments.map((segment, index) => {
          const href = '/' + segments.slice(0, index + 1).join('/')
          const isLast = index === segments.length - 1
          const label = segment.charAt(0).toUpperCase() + segment.slice(1)
          
          return (
            <React.Fragment key={segment}>
              <BreadcrumbSeparator />
              <BreadcrumbItem>
                {isLast ? (
                  <BreadcrumbPage>{label}</BreadcrumbPage>
                ) : (
                  <BreadcrumbLink href={href}>{label}</BreadcrumbLink>
                )}
              </BreadcrumbItem>
            </React.Fragment>
          )
        })}
      </BreadcrumbList>
    </Breadcrumb>
  )
}
```

## Best Practices

1. **Keep it simple**: Don't include every level of navigation
2. **Use meaningful labels**: Make breadcrumb text descriptive
3. **Current page**: Always show the current page as non-clickable
4. **Responsive design**: Consider mobile experience with ellipsis
5. **Consistent styling**: Match your site's navigation patterns
6. **Performance**: Avoid unnecessary re-renders with dynamic breadcrumbs
7. **Accessibility**: Test with screen readers and keyboard navigation

## Integration Examples

### With React Router

```tsx
import { useLocation, Link } from 'react-router-dom'

function RouterBreadcrumb() {
  const location = useLocation()
  const pathnames = location.pathname.split('/').filter(x => x)
  
  return (
    <Breadcrumb>
      <BreadcrumbList>
        <BreadcrumbItem>
          <BreadcrumbLink asChild>
            <Link to="/">Home</Link>
          </BreadcrumbLink>
        </BreadcrumbItem>
        {pathnames.map((name, index) => {
          const routeTo = `/${pathnames.slice(0, index + 1).join('/')}`
          const isLast = index === pathnames.length - 1
          
          return (
            <React.Fragment key={name}>
              <BreadcrumbSeparator />
              <BreadcrumbItem>
                {isLast ? (
                  <BreadcrumbPage>{name}</BreadcrumbPage>
                ) : (
                  <BreadcrumbLink asChild>
                    <Link to={routeTo}>{name}</Link>
                  </BreadcrumbLink>
                )}
              </BreadcrumbItem>
            </React.Fragment>
          )
        })}
      </BreadcrumbList>
    </Breadcrumb>
  )
}
```

### With Remix

```tsx
import { useMatches, Link } from '@remix-run/react'

function RemixBreadcrumb() {
  const matches = useMatches()
  
  return (
    <Breadcrumb>
      <BreadcrumbList>
        {matches.map((match, index) => {
          const isLast = index === matches.length - 1
          
          return (
            <React.Fragment key={match.id}>
              {index > 0 && <BreadcrumbSeparator />}
              <BreadcrumbItem>
                {isLast ? (
                  <BreadcrumbPage>{match.data?.title}</BreadcrumbPage>
                ) : (
                  <BreadcrumbLink asChild>
                    <Link to={match.pathname}>{match.data?.title}</Link>
                  </BreadcrumbLink>
                )}
              </BreadcrumbItem>
            </React.Fragment>
          )
        })}
      </BreadcrumbList>
    </Breadcrumb>
  )
}
```
