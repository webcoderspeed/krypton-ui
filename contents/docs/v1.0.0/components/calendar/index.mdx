# Calendar

A flexible calendar component for date selection with full keyboard navigation and accessibility support. Built on top of React DayPicker with customizable styling.

## Installation

```bash
npm install react-day-picker
```

## Import

```tsx
import { Calendar } from 'krypton-ui'
```

## Basic Usage

The Calendar component provides an intuitive interface for date selection.

<CodePreview
  code={`import { Calendar } from 'krypton-ui'
import { useState } from 'react'

export default function BasicCalendar() {
  const [date, setDate] = useState<Date | undefined>(new Date())

    <Calendar
      mode="single"

      onSelect={setDate}

## Date Range Selection

Select a range of dates for booking systems or date filters.

<CodePreview
  code={`import { Calendar } from 'krypton-ui'
import { useState } from 'react'
import { DateRange } from 'react-day-picker'

export default function DateRangeCalendar() {
  const [range, setRange] = useState<DateRange | undefined>({
    from: new Date(),
    to: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000)

  return (
    <Calendar
      mode="range"

      numberOfMonths={2}
      className="rounded-md border"
    />
  )
}`}

      }}
      onSelect={() => {}}
      numberOfMonths={2}
      className="rounded-md border"
    />
  }
/>

## Custom Footer

Add custom content to the calendar footer.

<CodePreview
  code={`import { Calendar } from 'krypton-ui'
import { Button } from 'krypton-ui'
import { useState } from 'react'

export default function CustomFooterCalendar() {
  const [date, setDate] = useState<Date | undefined>()

  const footer = date ? (
    <div className="flex justify-between items-center p-3 border-t">

        size="sm"

        Clear
      </Button>
    </div>

    <p className="text-sm text-muted-foreground p-3 border-t">
      Please pick a date.
    </p>
  )

  return (
    <Calendar
      mode="single"
      selected={date}
      onSelect={setDate}
      footer={footer}
      className="rounded-md border"
    />
  )
}`}
  preview={

      onSelect={() => {}}

            size="sm"

## API Reference

### Calendar

The main calendar component.

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| mode | "single" \| "multiple" \| "range" | "single" | The selection mode |
| selected | Date \| Date[] \| DateRange \| undefined | - | The selected date(s) |
| onSelect | (date: Date \| Date[] \| DateRange \| undefined) => void | - | Callback when selection changes |
| defaultMonth | Date | new Date() | The month to display initially |
| numberOfMonths | number | 1 | Number of months to display |
| fromDate | Date | - | The earliest date that can be selected |
| toDate | Date | - | The latest date that can be selected |
| disabled | Date \| Date[] \| DateRange \| ((date: Date) => boolean) | - | Dates that cannot be selected |
| footer | ReactNode | - | Custom footer content |
| captionLayout | "buttons" \| "dropdown" \| "dropdown-buttons" | "buttons" | Layout for the month/year navigation |
| fromYear | number | - | The earliest year for dropdown navigation |
| toYear | number | - | The latest year for dropdown navigation |
| className | string | - | Additional CSS classes |
| classNames | object | - | Custom class names for calendar parts |

## Common Patterns

### Date Picker with Input

```tsx
import { Calendar, Button, Popover, PopoverContent, PopoverTrigger, LucideIcon } from 'krypton-ui'
import { format } from 'date-fns'

function DatePicker() {
  const [date, setDate] = useState<Date>()

  return (
    <Popover>
      <PopoverTrigger asChild>
        <Button
          variant="outline"
          className="w-[280px] justify-start text-left font-normal"
        >
          <LucideIcon name="Calendar" className="mr-2 h-4 w-4" />
          {date ? format(date, "PPP") : <span>Pick a date</span>}
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-auto p-0">
        <Calendar
          mode="single"
          selected={date}
          onSelect={setDate}
          initialFocus
        />
      </PopoverContent>
    </Popover>
  )
}
```

### Booking Calendar

```tsx
function BookingCalendar() {
  const [range, setRange] = useState<DateRange | undefined>()
  
  const bookedDates = [
    new Date(2024, 0, 15),
    new Date(2024, 0, 16),
    { from: new Date(2024, 0, 20), to: new Date(2024, 0, 25) }
  ]

  return (
    <Calendar
      mode="range"
      selected={range}
      onSelect={setRange}
      disabled={bookedDates}
      numberOfMonths={2}
      className="rounded-md border"
    />
  )
}
```

## Troubleshooting

### Common Issues

1. **Date not updating**: Ensure you're using the `onSelect` callback to update state
2. **Styling issues**: Check that Tailwind CSS classes are properly configured
3. **Accessibility warnings**: Verify that proper labels and ARIA attributes are set
4. **Performance with large date ranges**: Consider using `fromDate` and `toDate` to limit the range
