### API

<CodePreview 
  code={`"use client"

import * as React from "react"
import {
  Carousel,
  CarouselContent,
  CarouselItem,
  CarouselNext,
  CarouselPrevious,
  type CarouselApi,
} from "@/components/ui/carousel"
import { Card, CardContent } from "@/components/ui/card"

export function CarouselApiDemo() {
  const [api, setApi] = React.useState<CarouselApi>()
  const [current, setCurrent] = React.useState(0)
  const [count, setCount] = React.useState(0)

  React.useEffect(() => {
    if (!api) {
      return
    }

    setCount(api.scrollSnapList().length)
    setCurrent(api.selectedScrollSnap() + 1)

    api.on("select", () => {
      setCurrent(api.selectedScrollSnap() + 1)
    })
  }, [api])

  return (
    <div>
      <Carousel setApi={setApi} className="w-full max-w-xs">
        <CarouselContent>
          {Array.from({ length: 5 }).map((_, index) => (
            <CarouselItem key={index}>
              <Card>
                <CardContent className="flex aspect-square items-center justify-center p-6">
                  <span className="text-4xl font-semibold">{index + 1}</span>
                </CardContent>
              </Card>
            </CarouselItem>
          ))}
        </CarouselContent>
        <CarouselPrevious />
        <CarouselNext />
      </Carousel>
      <div className="py-2 text-center text-sm text-muted-foreground">
        Slide {current} of {count}
      </div>
    </div>
  )
}`}

## API Reference

### Carousel

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | The orientation of the carousel |
| `opts` | `CarouselOptions` | - | Options to pass to the underlying Embla Carousel |
| `plugins` | `CarouselPlugin` | - | Plugins to pass to the underlying Embla Carousel |
| `setApi` | `(api: CarouselApi) => void` | - | Callback to set the carousel API instance |

### CarouselContent

Extends `div` element props.

### CarouselItem

Extends `div` element props.

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `className` | `string` | - | Additional CSS classes for the carousel item |

### CarouselPrevious

Extends `Button` component props.

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `variant` | `ButtonVariant` | `"outline"` | The variant of the previous button |
| `size` | `ButtonSize` | `"icon"` | The size of the previous button |

### CarouselNext

Extends `Button` component props.

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `variant` | `ButtonVariant` | `"outline"` | The variant of the next button |
| `size` | `ButtonSize` | `"icon"` | The size of the next button |

- Supports both horizontal and vertical orientations
