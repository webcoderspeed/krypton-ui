# Dialog

A window overlaid on either the primary window or another dialog window, rendering the content underneath inert. Perfect for confirmations, forms, or any content that requires user attention.

## Installation

```bash
npm install @radix-ui/react-dialog
npm install lucide-react
```

## Import

```tsx
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
  DialogFooter,
  DialogClose,
} from "@/components/ui/dialog"
```

### Controlled Dialog

```tsx
import { useState } from "react"
import { Button } from "@/components/ui/button"
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
  DialogFooter,
} from "@/components/ui/dialog"

export function ControlledDialog() {
  const [open, setOpen] = useState(false)

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button variant="outline">Open Dialog</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Are you absolutely sure?</DialogTitle>
          <DialogDescription>
            This action cannot be undone. This will permanently delete your
            account and remove your data from our servers.
          </DialogDescription>
        </DialogHeader>
        <DialogFooter>
          <Button variant="outline" onClick={() => setOpen(false)}>
            Cancel
          </Button>
          <Button variant="destructive" onClick={() => setOpen(false)}>
            Delete Account
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```

### Confirmation Dialog

```tsx
import { useState } from "react"
import { Trash2 } from "lucide-react"
import { Button } from "@/components/ui/button"
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
  DialogFooter,
} from "@/components/ui/dialog"

export function ConfirmationDialog() {
  const [open, setOpen] = useState(false)
  const [isDeleting, setIsDeleting] = useState(false)

  const handleDelete = async () => {
    setIsDeleting(true)
    // Simulate API call
    await new Promise(resolve => setTimeout(resolve, 2000))
    setIsDeleting(false)
    setOpen(false)
  }

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button variant="destructive" size="sm">
          <Trash2 className="h-4 w-4 mr-2" />
          Delete
        </Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Delete Item</DialogTitle>
          <DialogDescription>
            Are you sure you want to delete this item? This action cannot be undone.
          </DialogDescription>
        </DialogHeader>
        <DialogFooter>
          <Button
            variant="outline"
            onClick={() => setOpen(false)}
            disabled={isDeleting}
          >
            Cancel
          </Button>
          <Button
            variant="destructive"
            onClick={handleDelete}
            disabled={isDeleting}
          >
            {isDeleting ? "Deleting..." : "Delete"}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```

### Form Dialog

```tsx
import { useState } from "react"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { Textarea } from "@/components/ui/textarea"
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
  DialogFooter,
} from "@/components/ui/dialog"

export function FormDialog() {
  const [open, setOpen] = useState(false)
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    message: "",
  })

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    console.log("Form submitted:", formData)
    setOpen(false)
    setFormData({ name: "", email: "", message: "" })
  }

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button>Contact Us</Button>
      </DialogTrigger>
      <DialogContent className="sm:max-w-[425px]">
        <form onSubmit={handleSubmit}>
          <DialogHeader>
            <DialogTitle>Contact Us</DialogTitle>
            <DialogDescription>
              Send us a message and we'll get back to you as soon as possible.
            </DialogDescription>
          </DialogHeader>
          <div className="grid gap-4 py-4">
            <div className="grid gap-2">
              <Label htmlFor="name">Name</Label>
              <Input
                id="name"
                value={formData.name}
                onChange={(e) => setFormData(prev => ({ ...prev, name: e.target.value }))}
                placeholder="Your name"
                required
              />
            </div>
            <div className="grid gap-2">
              <Label htmlFor="email">Email</Label>
              <Input
                id="email"
                type="email"
                value={formData.email}
                onChange={(e) => setFormData(prev => ({ ...prev, email: e.target.value }))}
                placeholder="your@email.com"
                required
              />
            </div>
            <div className="grid gap-2">
              <Label htmlFor="message">Message</Label>
              <Textarea
                id="message"
                value={formData.message}
                onChange={(e) => setFormData(prev => ({ ...prev, message: e.target.value }))}
                placeholder="Your message..."
                required
              />
            </div>
          </div>
          <DialogFooter>
            <Button type="button" variant="outline" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button type="submit">Send Message</Button>
          </DialogFooter>
        </form>
      </DialogContent>
    </Dialog>
  )
}
```

## API Reference

### Dialog

The root container for the dialog.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | - | The controlled open state |
| `defaultOpen` | `boolean` | `false` | The default open state (uncontrolled) |
| `onOpenChange` | `(open: boolean) => void` | - | Callback when open state changes |
| `modal` | `boolean` | `true` | Whether the dialog is modal |
| `children` | `ReactNode` | - | The dialog content |

### DialogTrigger

The button that opens the dialog.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | `false` | Render as child component |
| `children` | `ReactNode` | - | The trigger content |

### DialogContent

The dialog content container.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `forceMount` | `boolean` | `false` | Force mount the content |
| `onOpenAutoFocus` | `(event: Event) => void` | - | Callback when dialog opens and focuses |
| `onCloseAutoFocus` | `(event: Event) => void` | - | Callback when dialog closes and focuses |
| `onEscapeKeyDown` | `(event: KeyboardEvent) => void` | - | Callback when escape key is pressed |
| `onPointerDownOutside` | `(event: PointerEvent) => void` | - | Callback when pointer is pressed outside |
| `onInteractOutside` | `(event: Event) => void` | - | Callback when interaction happens outside |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The content |

**Default Styles:**
- `fixed left-[50%] top-[50%]` - Centered positioning
- `z-50 grid w-full max-w-lg` - Z-index and layout
- `translate-x-[-50%] translate-y-[-50%]` - Center transform
- `gap-4 border bg-background p-6` - Spacing and appearance
- `shadow-lg duration-200` - Visual effects
- `data-[state=open]:animate-in` - Opening animation
- `data-[state=closed]:animate-out` - Closing animation
- `sm:rounded-lg` - Responsive border radius

### DialogHeader

Container for dialog title and description.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The header content |

**Default Styles:**
- `flex flex-col space-y-1.5` - Vertical layout
- `text-center sm:text-left` - Responsive text alignment

### DialogTitle

The dialog title.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The title content |

**Default Styles:**
- `text-lg font-semibold leading-none tracking-tight` - Typography

### DialogDescription

The dialog description.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The description content |

**Default Styles:**
- `text-sm text-muted-foreground` - Typography and color

### DialogFooter

Container for dialog actions.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The footer content |

**Default Styles:**
- `flex flex-col-reverse sm:flex-row` - Responsive layout
- `sm:justify-end sm:space-x-2` - Responsive spacing

### DialogClose

A button that closes the dialog.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | `false` | Render as child component |
| `children` | `ReactNode` | - | The close button content |

## Features

- **Accessible**: Full keyboard navigation and screen reader support
- **Customizable**: Easy to style and extend
- **Animated**: Smooth open/close animations
- **Focus Management**: Automatic focus handling
- **Escape Key**: Close with escape key
- **Click Outside**: Close when clicking outside (configurable)
- **Portal Rendering**: Renders in a portal by default
- **Controlled/Uncontrolled**: Support for both usage patterns

## Accessibility

- Uses proper ARIA attributes (`role="dialog"`, `aria-labelledby`, `aria-describedby`)
- Focus is trapped within the dialog when open
- Focus returns to trigger when closed
- Escape key closes the dialog
- Background is inert when dialog is open
- Screen reader announcements for state changes

5. **Z-index**: Proper layering

## Common Patterns

### Multi-step Dialog

```tsx
function MultiStepDialog() {
  const [step, setStep] = useState(1)
  const [open, setOpen] = useState(false)

  const nextStep = () => setStep(prev => prev + 1)
  const prevStep = () => setStep(prev => prev - 1)
  const resetAndClose = () => {
    setStep(1)
    setOpen(false)
  }

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button>Start Setup</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Setup Wizard - Step {step} of 3</DialogTitle>
          <DialogDescription>
            {step === 1 && "Let's start with your basic information."}
            {step === 2 && "Now, let's configure your preferences."}
            {step === 3 && "Finally, review your settings."}
          </DialogDescription>
        </DialogHeader>
        
        {/* Step content */}
        <div className="py-4">
          {step === 1 && <div>Step 1 content...</div>}
          {step === 2 && <div>Step 2 content...</div>}
          {step === 3 && <div>Step 3 content...</div>}
        </div>
        
        <DialogFooter>
          {step > 1 && (
            <Button variant="outline" onClick={prevStep}>
              Previous
            </Button>
          )}
          {step < 3 ? (
            <Button onClick={nextStep}>Next</Button>
          ) : (
            <Button onClick={resetAndClose}>Finish</Button>
          )}
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```

## Best Practices

1. **Clear Purpose**: Each dialog should have a single, clear purpose
2. **Descriptive Titles**: Use clear, descriptive titles
3. **Escape Routes**: Always provide a way to close the dialog
4. **Focus Management**: Ensure proper focus handling
5. **Mobile Friendly**: Test on mobile devices
6. **Loading States**: Handle async operations gracefully
7. **Validation**: Validate forms before submission
8. **Confirmation**: Use confirmation dialogs for destructive actions
9. **Size Appropriately**: Choose appropriate sizes for content
10. **Accessibility**: Test with screen readers and keyboard navigation

## Troubleshooting

### Dialog Not Opening

Ensure the trigger is properly configured:

```tsx
<DialogTrigger asChild>
  <Button>Open Dialog</Button>
</DialogTrigger>
```

### Focus Issues

If focus management isn't working, check for:
- Conflicting focus management libraries
- Custom focus handling that interferes
- Missing `asChild` prop on triggers

### Styling Issues

Ensure Tailwind CSS is properly configured and the component styles are included in your build.

### Animation Problems

If animations aren't working:
- Check that Tailwind animations are enabled
- Verify CSS custom properties are supported
- Test in different browsers
