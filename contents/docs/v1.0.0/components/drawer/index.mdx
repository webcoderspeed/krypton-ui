### Controlled Drawer

<CodePreview
  code={`import {
  Button,
  Drawer,
  DrawerClose,
  DrawerContent,
  DrawerDescription,
  DrawerFooter,
  DrawerHeader,
  DrawerTitle,
  DrawerTrigger,
} from "krypton-ui"
import { useState } from "react"

export default function Component() {
  const [open, setOpen] = useState(false)

  return (
    <div className="space-x-2">
      <Drawer open={open} onOpenChange={setOpen}>
        <DrawerTrigger asChild>
          <Button variant="outline">Open Controlled Drawer</Button>
        </DrawerTrigger>
        <DrawerContent>
          <DrawerHeader>
            <DrawerTitle>Controlled Drawer</DrawerTitle>
            <DrawerDescription>
              This drawer's open state is controlled by React state.
            </DrawerDescription>
          </DrawerHeader>
          <div className="p-4 pb-0">
            <p className="text-sm text-muted-foreground">
              You can control this drawer programmatically using the buttons below.
            </p>
          </div>
          <DrawerFooter>
            <Button onClick={() => setOpen(false)}>Close Programmatically</Button>
            <DrawerClose asChild>
              <Button variant="outline">Close with DrawerClose</Button>
            </DrawerClose>
          </DrawerFooter>
        </DrawerContent>
      </Drawer>
      <Button 
        variant="secondary" 
        onClick={() => setOpen(true)}
        disabled={open}
      >
        Open Externally
      </Button>
    </div>
  )
}`}
  preview={
    <div className="space-x-2">
      <Drawer>
        <DrawerTrigger asChild>
          <Button variant="outline">Open Controlled Drawer</Button>
        </DrawerTrigger>
        <DrawerContent>
          <DrawerHeader>
            <DrawerTitle>Controlled Drawer</DrawerTitle>
            <DrawerDescription>
              This drawer's open state is controlled by React state.
            </DrawerDescription>
          </DrawerHeader>
          <div className="p-4 pb-0">
            <p className="text-sm text-muted-foreground">
              You can control this drawer programmatically using the buttons below.
            </p>
          </div>
          <DrawerFooter>
            <Button>Close Programmatically</Button>
            <DrawerClose asChild>
              <Button variant="outline">Close with DrawerClose</Button>
            </DrawerClose>
          </DrawerFooter>
        </DrawerContent>
      </Drawer>
      <Button variant="secondary">
        Open Externally
      </Button>
    </div>
  }
/>

## API Reference

### Drawer

The root component that manages the drawer state.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `open` | `boolean` | - | The controlled open state of the drawer |
| `onOpenChange` | `(open: boolean) => void` | - | Event handler called when the open state changes |
| `shouldScaleBackground` | `boolean` | `true` | Whether the background should scale when the drawer is open |
| `children` | `ReactNode` | - | The drawer trigger and content |

### DrawerTrigger

The button that opens the drawer.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | `false` | Change the default rendered element |
| `children` | `ReactNode` | - | The trigger content |
| `className` | `string` | - | Additional CSS classes |

### DrawerContent

The component that contains the drawer content.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The drawer content |

### DrawerHeader

The header section of the drawer.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The header content |

### DrawerTitle

The title of the drawer.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The title content |

### DrawerDescription

The description text of the drawer.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The description content |

### DrawerFooter

The footer section of the drawer.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The footer content |

### DrawerClose

A button that closes the drawer.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | `false` | Change the default rendered element |
| `children` | `ReactNode` | - | The close button content |
| `className` | `string` | - | Additional CSS classes |
