    setValue("");

        <CardTitle>Verify Your Account</CardTitle>

### Email Verification

Email verification flow with countdown timer and resend functionality.

<CodePreview

import {
  InputOTP,
  InputOTPGroup,
  InputOTPSeparator,
  InputOTPSlot,

export default function InputOTPEmailVerification() {
  const [value, setValue] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const [status, setStatus] = useState<"idle" | "success" | "error">("idle");
  const [countdown, setCountdown] = useState(60);
  const [canResend, setCanResend] = useState(false);

  useEffect(() => {
    if (countdown > 0) {
      const timer = setTimeout(() => setCountdown(countdown - 1), 1000);
      return () => clearTimeout(timer);
    } else {
      setCanResend(true);
    }
  }, [countdown]);

  const handleVerify = async () => {
    if (value.length !== 6) return;
    
    setIsLoading(true);
    setStatus("idle");
    
    // Simulate API call
    await new Promise(resolve => setTimeout(resolve, 2000));
    
    if (value === "123456") {
      setStatus("success");
    } else {
      setStatus("error");
    }
    setIsLoading(false);
  };

  const handleResend = () => {
    setValue("");
    setStatus("idle");
    setCountdown(60);
    setCanResend(false);
    // Simulate resend logic
    console.log("Resending verification email...");
  };

  const progressValue = ((60 - countdown) / 60) * 100;

  return (
    <Card className="w-full max-w-md mx-auto">
      <CardHeader className="text-center">
        <div className="mx-auto w-12 h-12 bg-blue-100 rounded-full flex items-center justify-center mb-4">

        </div>
        <CardTitle>Verify Your Email</CardTitle>

          <strong>john.doe@example.com</strong>
        </CardDescription>
      </CardHeader>
      <CardContent className="space-y-4">
        <div className="space-y-2">
          <label className="text-sm font-medium">Verification Code</label>
          <div className="flex justify-center">
            <InputOTP
              maxLength={6}
              value={value}
              onChange={(value) => setValue(value)}
            >
              <InputOTPGroup>
                <InputOTPSlot index={0} />
                <InputOTPSlot index={1} />
                <InputOTPSlot index={2} />
              </InputOTPGroup>
              <InputOTPSeparator />
              <InputOTPGroup>
                <InputOTPSlot index={3} />
                <InputOTPSlot index={4} />
                <InputOTPSlot index={5} />
              </InputOTPGroup>
            </InputOTP>
          </div>
        </div>

        {!canResend && (
          <div className="space-y-2">
            <div className="flex items-center gap-2 text-sm text-muted-foreground">
              <Clock className="h-4 w-4" />
              <span>Resend available in {countdown}s</span>
            </div>
            <Progress value={progressValue} className="h-2" />
          </div>
        )}

        {status === "success" && (

  );

## API Reference

### InputOTP

The main container component for the OTP input.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `maxLength` | `number` | `6` | Maximum number of characters |
| `value` | `string` | - | Controlled value of the input |
| `onChange` | `(value: string) => void` | - | Callback when value changes |
| `onComplete` | `(value: string) => void` | - | Callback when all slots are filled |
| `disabled` | `boolean` | `false` | Whether the input is disabled |
| `autoFocus` | `boolean` | `false` | Whether to auto-focus the first slot |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | InputOTPGroup and InputOTPSeparator components |

### InputOTPGroup

Groups multiple InputOTPSlot components together.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | InputOTPSlot components |

### InputOTPSlot

Individual input slot for a single character.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `index` | `number` | - | The index of this slot (required) |
| `className` | `string` | - | Additional CSS classes |

### InputOTPSeparator

Visual separator between groups of slots.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |

## Accessibility

The InputOTP component follows accessibility best practices:

### Keyboard Support

| Key | Description |
|-----|-------------|
| `Tab` | Moves focus to the next focusable element |
| `Shift + Tab` | Moves focus to the previous focusable element |
| `Arrow Left` | Moves focus to the previous slot |
| `Arrow Right` | Moves focus to the next slot |
| `Backspace` | Deletes the current character and moves to previous slot |
| `Delete` | Deletes the current character |
| `Home` | Moves focus to the first slot |
| `End` | Moves focus to the last slot |

### Screen Reader Support

- Each slot is properly labeled for screen readers
- The component announces the current position and total length
- Completion status is announced when all slots are filled
- Error states are properly communicated

### Focus Management

- Focus moves automatically between slots as user types
- Focus is trapped within the OTP input when active
- Visual focus indicators are clearly visible
- Supports both mouse and keyboard interactions

## Best Practices

1. **Use appropriate length**: Choose the right number of digits for your use case (4-8 digits are common)
2. **Provide clear instructions**: Always include labels and descriptions
3. **Handle errors gracefully**: Show clear error messages for invalid codes
4. **Auto-advance focus**: Let users type continuously without manual navigation
5. **Support paste**: Allow users to paste complete codes
6. **Add visual separators**: Use separators for better readability with longer codes
7. **Implement timeouts**: Add expiration times for security codes
8. **Provide resend options**: Allow users to request new codes if needed
