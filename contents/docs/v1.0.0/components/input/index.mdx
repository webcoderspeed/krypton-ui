# Input

Displays a form input field or a component that looks like an input field. Perfect for collecting user data in forms, search boxes, and other interactive elements.

## Installation

```bash
npm install react
```

## Import

```tsx
import { Input } from "@/components/ui/input"
```

```tsx
          ) : (
            <Eye className="h-4 w-4" aria-hidden="true" />

          </span>
        </Button>

  )
```

### Form Example

```tsx
import { useState } from "react"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"

export function FormExample() {
  const [formData, setFormData] = useState({
    firstName: '',
    lastName: '',
    email: '',
    phone: '',
  })

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    console.log('Form submitted:', formData)
  }

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    setFormData(prev => ({
      ...prev,
      [e.target.name]: e.target.value
    }))
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-4 max-w-sm">
      <div className="grid grid-cols-2 gap-4">
        <div className="grid items-center gap-1.5">
          <Label htmlFor="firstName">First Name</Label>
          <Input
            id="firstName"
            name="firstName"
            value={formData.firstName}
            onChange={handleChange}
            placeholder="John"
            required
          />
        </div>
        <div className="grid items-center gap-1.5">
          <Label htmlFor="lastName">Last Name</Label>
          <Input
            id="lastName"
            name="lastName"
            value={formData.lastName}
            onChange={handleChange}
            placeholder="Doe"
            required
          />
        </div>
      </div>
      
      <div className="grid items-center gap-1.5">
        <Label htmlFor="email">Email</Label>
        <Input
          id="email"
          name="email"
          type="email"
          value={formData.email}
          onChange={handleChange}
          placeholder="john@example.com"
          required
        />
      </div>
      
      <div className="grid items-center gap-1.5">
        <Label htmlFor="phone">Phone</Label>
        <Input
          id="phone"
          name="phone"
          type="tel"
          value={formData.phone}
          onChange={handleChange}
          placeholder="+1 (555) 123-4567"
        />
      </div>
      
      <Button type="submit" className="w-full">
        Submit
      </Button>
    </form>
  )
}
```

### Input with Validation

```tsx
import { useState } from "react"
import { Input } from "@/components/ui/input"
import { Label } from "@/components/ui/label"
import { CheckCircle, XCircle } from "lucide-react"

export function InputWithValidation() {
  const [email, setEmail] = useState("")
  const [isValid, setIsValid] = useState<boolean | null>(null)

  const validateEmail = (email: string) => {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
    return emailRegex.test(email)
  }

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value
    setEmail(value)
    
    if (value === "") {
      setIsValid(null)
    } else {
      setIsValid(validateEmail(value))
    }
  }

  return (
    <div className="grid w-full max-w-sm items-center gap-1.5">
      <Label htmlFor="email-validation">Email</Label>
      <div className="relative">
        <Input
          id="email-validation"
          type="email"
          value={email}
          onChange={handleChange}
          placeholder="Enter your email"
          className={`pr-10 ${
            isValid === false
              ? "border-destructive focus-visible:ring-destructive"
              : isValid === true
              ? "border-green-500 focus-visible:ring-green-500"
              : ""
          }`}
        />
        {isValid !== null && (
          <div className="absolute right-3 top-2.5">
            {isValid ? (
              <CheckCircle className="h-4 w-4 text-green-500" />
            ) : (
              <XCircle className="h-4 w-4 text-destructive" />
            )}
          </div>
        )}
      </div>
      {isValid === false && (
        <p className="text-sm text-destructive">
          Please enter a valid email address.
        </p>
      )}
      {isValid === true && (
        <p className="text-sm text-green-600">
          Email looks good!
        </p>
      )}
    </div>
  )
}
```

## API Reference

### Input

Extends the standard HTML `input` element with additional styling and functionality.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `string` | `"text"` | The input type (text, email, password, etc.) |
| `placeholder` | `string` | - | Placeholder text |
| `value` | `string` | - | The input value (controlled) |
| `defaultValue` | `string` | - | The default value (uncontrolled) |
| `onChange` | `(event: ChangeEvent<HTMLInputElement>) => void` | - | Change event handler |
| `onFocus` | `(event: FocusEvent<HTMLInputElement>) => void` | - | Focus event handler |
| `onBlur` | `(event: FocusEvent<HTMLInputElement>) => void` | - | Blur event handler |
| `disabled` | `boolean` | `false` | Whether the input is disabled |
| `readOnly` | `boolean` | `false` | Whether the input is read-only |
| `required` | `boolean` | `false` | Whether the input is required |
| `autoComplete` | `string` | - | Autocomplete attribute |
| `autoFocus` | `boolean` | `false` | Whether to auto-focus the input |
| `name` | `string` | - | The input name |
| `id` | `string` | - | The input ID |
| `className` | `string` | - | Additional CSS classes |
| `min` | `string \| number` | - | Minimum value (for number/date inputs) |
| `max` | `string \| number` | - | Maximum value (for number/date inputs) |
| `step` | `string \| number` | - | Step value (for number inputs) |
| `pattern` | `string` | - | Validation pattern (regex) |
| `maxLength` | `number` | - | Maximum character length |
| `minLength` | `number` | - | Minimum character length |
| `size` | `number` | - | Input size attribute |
| `accept` | `string` | - | Accepted file types (for file inputs) |
| `multiple` | `boolean` | `false` | Allow multiple files (for file inputs) |
| `capture` | `string` | - | Camera capture mode (for file inputs) |

**Default Styles:**
- `flex h-10 w-full rounded-md border` - Layout and shape
- `border-input bg-background px-3 py-2` - Appearance and spacing
- `text-sm ring-offset-background` - Typography
- `file:border-0 file:bg-transparent file:text-sm file:font-medium` - File input styles
- `placeholder:text-muted-foreground` - Placeholder styling
- `focus-visible:outline-none focus-visible:ring-2` - Focus states
- `focus-visible:ring-ring focus-visible:ring-offset-2` - Focus ring
- `disabled:cursor-not-allowed disabled:opacity-50` - Disabled state

## Input Types

The Input component supports all standard HTML input types:

- `text` - Default text input
- `email` - Email input with validation
- `password` - Password input (hidden text)
- `number` - Numeric input
- `tel` - Telephone number input
- `url` - URL input
- `search` - Search input
- `date` - Date picker
- `time` - Time picker
- `datetime-local` - Date and time picker
- `month` - Month picker
- `week` - Week picker
- `color` - Color picker
- `file` - File upload
- `range` - Range slider
- `checkbox` - Checkbox (though use Checkbox component instead)
- `radio` - Radio button (though use RadioGroup component instead)
- `hidden` - Hidden input

## Best Practices

1. **Clear Labels**: Always provide clear, descriptive labels
2. **Placeholder Text**: Use helpful placeholder text that doesn't replace labels
3. **Validation**: Provide clear validation feedback
4. **Required Fields**: Mark required fields clearly
5. **Error Messages**: Show specific, actionable error messages
6. **Autocomplete**: Use appropriate autocomplete attributes
7. **Input Types**: Use the correct input type for the data
8. **Accessibility**: Ensure proper label association and ARIA attributes
9. **Mobile Friendly**: Test on mobile devices for usability
10. **Progressive Enhancement**: Ensure basic functionality without JavaScript

## Troubleshooting

### Input Not Focusing

Ensure the input is not disabled and check for CSS issues:

```tsx
<Input autoFocus /> // Auto-focus on mount
```

### Styling Issues

Ensure Tailwind CSS is properly configured:

```tsx
// Check if custom classes are being applied
<Input className="border-red-500" />
```

### Validation Not Working

Ensure proper form setup:

```tsx
<form onSubmit={handleSubmit}>
  <Input required type="email" />
  <Button type="submit">Submit</Button>
</form>
```

### Value Not Updating

For controlled inputs, ensure you're updating state:

```tsx
const [value, setValue] = useState("")

<Input
  value={value}
  onChange={(e) => setValue(e.target.value)}
/>
```
