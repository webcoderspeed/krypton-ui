# Scroll Area

Augments native scroll functionality for custom, cross-browser styling. Perfect for creating custom scrollbars that match your design system while maintaining native scroll behavior.

## Installation

```bash
npm install @radix-ui/react-scroll-area
```

## Import

```tsx
import { ScrollArea } from "@/components/ui/scroll-area"
```

## Usage

### Basic Example

```tsx
import { ScrollArea } from "@/components/ui/scroll-area"
import { Separator } from "@/components/ui/separator"

const tags = Array.from({ length: 50 }).map(
  (_, i, a) => `v1.2.0-beta.${a.length - i}`
)

export function ScrollAreaDemo() {
  return (
    <ScrollArea className="h-72 w-48 rounded-md border">
      <div className="p-4">
        <h4 className="mb-4 text-sm font-medium leading-none">Tags</h4>
        {tags.map((tag) => (
          <>
            <div key={tag} className="text-sm">
              {tag}
            </div>
            <Separator className="my-2" />
          </>
        ))}
      </div>
    </ScrollArea>
  )
}
```

### File Explorer

```tsx
import { ScrollArea } from "@/components/ui/scroll-area"
import { Folder, File, ChevronRight, ChevronDown } from "lucide-react"
import { useState } from "react"

interface FileNode {
  name: string
  type: 'file' | 'folder'
  children?: FileNode[]
}

const fileStructure: FileNode[] = [
  {
    name: 'src',
    type: 'folder',
    children: [
      {
        name: 'components',
        type: 'folder',
        children: [
          { name: 'Button.tsx', type: 'file' },
          { name: 'Input.tsx', type: 'file' },
          { name: 'ScrollArea.tsx', type: 'file' },
        ],
      },
      {
        name: 'pages',
        type: 'folder',
        children: [
          { name: 'Home.tsx', type: 'file' },
          { name: 'About.tsx', type: 'file' },
        ],
      },
      { name: 'App.tsx', type: 'file' },
      { name: 'index.tsx', type: 'file' },
    ],
  },
  {
    name: 'public',
    type: 'folder',
    children: [
      { name: 'index.html', type: 'file' },
      { name: 'favicon.ico', type: 'file' },
    ],
  },
  { name: 'package.json', type: 'file' },
  { name: 'README.md', type: 'file' },
]

function FileTreeNode({ node, depth = 0 }: { node: FileNode; depth?: number }) {
  const [isOpen, setIsOpen] = useState(false)

  return (
    <div>
      <div
        className={`flex items-center py-1 px-2 hover:bg-accent rounded-sm cursor-pointer`}
        style={{ paddingLeft: `${depth * 16 + 8}px` }}
        onClick={() => node.type === 'folder' && setIsOpen(!isOpen)}
      >
        {node.type === 'folder' ? (
          <>
            {isOpen ? (
              <ChevronDown className="h-4 w-4 mr-1" />
            ) : (
              <ChevronRight className="h-4 w-4 mr-1" />
            )}
            <Folder className="h-4 w-4 mr-2 text-blue-500" />
          </>
        ) : (
          <>
            <div className="w-5" />
            <File className="h-4 w-4 mr-2 text-gray-500" />
          </>
        )}
        <span className="text-sm">{node.name}</span>
      </div>
      {node.type === 'folder' && isOpen && node.children && (
        <div>
          {node.children.map((child, index) => (
            <FileTreeNode key={index} node={child} depth={depth + 1} />
          ))}
        </div>
      )}
    </div>
  )
}

export function FileExplorerScrollArea() {
  return (
    <ScrollArea className="h-80 w-64 rounded-md border">
      <div className="p-2">
        <h4 className="mb-2 text-sm font-medium">Project Files</h4>
        {fileStructure.map((node, index) => (
          <FileTreeNode key={index} node={node} />
        ))}
      </div>
    </ScrollArea>
  )
}
```

## API Reference

### ScrollArea

The main scroll area container.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `type` | `"auto" \| "always" \| "scroll" \| "hover"` | `"hover"` | When to show scrollbars |
| `scrollHideDelay` | `number` | `600` | Hide delay in milliseconds |
| `dir` | `"ltr" \| "rtl"` | - | Reading direction |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The scrollable content |

**Default Styles:**
- `relative overflow-hidden` - Layout and overflow handling

### ScrollBar

The scrollbar component (optional, for horizontal scrolling).

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"vertical" \| "horizontal"` | `"vertical"` | Scrollbar orientation |
| `className` | `string` | - | Additional CSS classes |

**Default Styles:**
- `flex touch-none select-none transition-colors` - Layout and interaction
- `data-[orientation=vertical]:h-full data-[orientation=vertical]:w-2.5` - Vertical sizing
- `data-[orientation=horizontal]:h-2.5 data-[orientation=horizontal]:w-full` - Horizontal sizing
- `data-[orientation=horizontal]:flex-col` - Horizontal layout

## Scroll Types

- **`auto`**: Scrollbars appear automatically when content overflows
- **`always`**: Scrollbars are always visible
- **`scroll`**: Scrollbars are always visible and functional
- **`hover`**: Scrollbars appear on hover (default)

## Best Practices

1. **Container Height**: Always set a specific height for the scroll area
2. **Content Padding**: Add appropriate padding to scrollable content
3. **Horizontal Scrolling**: Include `ScrollBar` component for horizontal scroll
4. **Performance**: Avoid nesting multiple scroll areas
5. **Mobile**: Test touch scrolling on mobile devices
6. **Accessibility**: Ensure keyboard navigation works properly
7. **Visual Cues**: Provide clear indication of scrollable content
8. **Loading States**: Handle loading states gracefully
9. **Empty States**: Show appropriate empty states
10. **Responsive**: Consider different screen sizes

## Troubleshooting

### Scrollbar Not Appearing

Ensure the container has a fixed height:

```tsx
<ScrollArea className="h-96"> {/* Fixed height required */}
  {/* content */}
</ScrollArea>
```

### Horizontal Scroll Not Working

Include the ScrollBar component:

```tsx
<ScrollArea className="w-96">
  <div className="flex w-max space-x-4">
    {/* horizontal content */}
  </div>
  <ScrollBar orientation="horizontal" />
</ScrollArea>
```

### Content Not Scrolling

Check that content exceeds container dimensions:

```tsx
// Content must be larger than container
<ScrollArea className="h-48">
  <div className="h-96"> {/* Taller than container */}
    {/* content */}
  </div>
</ScrollArea>
```

### Styling Issues

Ensure proper CSS cascade and Tailwind configuration:

```tsx
// Use specific selectors for scrollbar styling
<ScrollArea className="[&>[data-radix-scroll-area-scrollbar]]:bg-accent">
  {/* content */}
</ScrollArea>
```
