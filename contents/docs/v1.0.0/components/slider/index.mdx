### Media Player Progress

Interactive progress bar for media playback control with play/pause functionality.

<CodePreview 
  code={`"use client"

import { useState, useEffect } from "react"
import { Slider } from "@/components/ui/slider"
import { Button } from "@/components/ui/button"
import { Play, Pause, SkipBack, SkipForward } from "lucide-react"

export default function SliderProgress() {
  const [progress, setProgress] = useState([0])
  const [isPlaying, setIsPlaying] = useState(false)
  const [duration] = useState(180) // 3 minutes

  useEffect(() => {
    let interval: NodeJS.Timeout
    if (isPlaying && progress[0] < duration) {
      interval = setInterval(() => {
        setProgress(prev => [Math.min(prev[0] + 1, duration)])
      }, 1000)
    }
    return () => clearInterval(interval)
  }, [isPlaying, progress, duration])

  const formatTime = (seconds: number) => {
    const mins = Math.floor(seconds / 60)
    const secs = seconds % 60
    return \`\${mins}:\${secs.toString().padStart(2, '0')}\`
  }

  const handlePlayPause = () => {
    setIsPlaying(!isPlaying)
  }

  const handleSkipBack = () => {
    setProgress([Math.max(progress[0] - 10, 0)])
  }

  const handleSkipForward = () => {
    setProgress([Math.min(progress[0] + 10, duration)])
  }

  return (
    <div className="w-full max-w-sm space-y-4 p-4 border rounded-lg">
      <div className="text-center">
        <h3 className="font-medium">Now Playing</h3>
        <p className="text-sm text-muted-foreground">Sample Track</p>
      </div>
      
      <div className="space-y-2">
        <Slider
          value={progress}
          onValueChange={setProgress}
          max={duration}
          min={0}
          step={1}
          className="w-full"
        />
        <div className="flex justify-between text-xs text-muted-foreground">
          <span>{formatTime(progress[0])}</span>
          <span>{formatTime(duration)}</span>
        </div>
      </div>

      <div className="flex items-center justify-center gap-2">
        <Button variant="outline" size="icon" onClick={handleSkipBack}>
          <SkipBack className="h-4 w-4" />
        </Button>
        <Button size="icon" onClick={handlePlayPause}>
          {isPlaying ? <Pause className="h-4 w-4" /> : <Play className="h-4 w-4" />}
        </Button>
        <Button variant="outline" size="icon" onClick={handleSkipForward}>
          <SkipForward className="h-4 w-4" />
        </Button>
      </div>
    </div>
  )
}`} 
  language="tsx" 
  preview={<SliderProgress />} 
/>

## API Reference

### Props

The Slider component accepts all props from Radix UI Slider primitive.

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `value` | `number[]` | `undefined` | The controlled value of the slider. |
| `defaultValue` | `number[]` | `undefined` | The default value when uncontrolled. |
| `onValueChange` | `(value: number[]) => void` | `undefined` | Callback fired when the value changes. |
| `onValueCommit` | `(value: number[]) => void` | `undefined` | Callback fired when the user stops dragging. |
| `min` | `number` | `0` | The minimum value of the slider. |
| `max` | `number` | `100` | The maximum value of the slider. |
| `step` | `number` | `1` | The step increment of the slider. |
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | The orientation of the slider. |
| `disabled` | `boolean` | `false` | Whether the slider is disabled. |
| `inverted` | `boolean` | `false` | Whether the slider is inverted. |
| `className` | `string` | `undefined` | Additional CSS classes. |

### Keyboard Navigation

| Key | Description |
| --- | --- |
| `ArrowLeft` / `ArrowDown` | Decrease the value by one step |
| `ArrowRight` / `ArrowUp` | Increase the value by one step |
| `PageDown` | Decrease the value by 10 steps |
| `PageUp` | Increase the value by 10 steps |
| `Home` | Set to minimum value |
| `End` | Set to maximum value |

## Accessibility

- Uses semantic HTML elements and ARIA attributes
- Supports keyboard navigation
- Screen reader compatible with proper labeling
- Focus management follows WAI-ARIA guidelines
- Supports both horizontal and vertical orientations

## Best Practices

### Do's
- Always provide clear labels for sliders
- Use appropriate min, max, and step values for your use case
- Consider adding visual indicators for important values
- Provide immediate feedback when values change
- Use range sliders for filtering scenarios

### Don'ts
- Don't use sliders for precise numeric input
- Avoid very large ranges without appropriate step values
- Don't forget to handle edge cases (min/max values)
- Avoid using sliders for binary choices (use Switch instead)
