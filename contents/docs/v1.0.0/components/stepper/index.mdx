# Stepper

A component that displays progress through a sequence of logical and numbered steps. Perfect for multi-step forms, onboarding flows, checkout processes, and guided workflows.

## Installation

No additional dependencies required. The stepper component is built using basic React and Tailwind CSS.

## Import

```tsx
import {
  Stepper,
  StepperItem,
  StepperSeparator,
  StepperTrigger,
  StepperContent,
} from "@/components/ui/stepper"
```

### Controlled Stepper

```tsx
import { useState } from "react"
import { Button } from "@/components/ui/button"
import {
  Stepper,
  StepperItem,
  StepperSeparator,
  StepperTrigger,
  StepperContent,
} from "@/components/ui/stepper"

const steps = [
  { id: "step1", title: "Personal Info", description: "Enter your details" },
  { id: "step2", title: "Account Setup", description: "Configure your account" },
  { id: "step3", title: "Verification", description: "Verify your email" },
  { id: "step4", title: "Complete", description: "Setup complete" },
]

export function ControlledStepper() {
  const [currentStep, setCurrentStep] = useState("step1")
  const currentIndex = steps.findIndex(step => step.id === currentStep)

  const nextStep = () => {
    const nextIndex = currentIndex + 1
    if (nextIndex < steps.length) {
      setCurrentStep(steps[nextIndex].id)
    }
  }

  const prevStep = () => {
    const prevIndex = currentIndex - 1
    if (prevIndex >= 0) {
      setCurrentStep(steps[prevIndex].id)
    }
  }

  return (
    <div className="w-full max-w-2xl mx-auto">
      <Stepper value={currentStep} onValueChange={setCurrentStep}>
        <div className="flex items-center">
          {steps.map((step, index) => (
            <div key={step.id} className="flex items-center">
              <StepperItem value={step.id}>
                <StepperTrigger className="w-10 h-10">
                  {index + 1}
                </StepperTrigger>
              </StepperItem>
              {index < steps.length - 1 && <StepperSeparator />}
            </div>
          ))}
        </div>
        
        <div className="mt-8">
          {steps.map((step) => (
            <StepperContent key={step.id} value={step.id}>
              <div className="space-y-4">
                <h3 className="text-lg font-semibold">{step.title}</h3>
                <p className="text-muted-foreground">{step.description}</p>
                
                <div className="flex justify-between pt-4">
                  <Button
                    variant="outline"
                    onClick={prevStep}
                    disabled={currentIndex === 0}
                  >
                    Previous
                  </Button>
                  <Button
                    onClick={nextStep}
                    disabled={currentIndex === steps.length - 1}
                  >
                    {currentIndex === steps.length - 1 ? "Complete" : "Next"}
                  </Button>
                </div>
              </div>
            </StepperContent>
          ))}
        </div>
      </Stepper>
    </div>
  )
}
```

### Progress Stepper

```tsx
import { useState, useEffect } from "react"
import { Button } from "@/components/ui/button"
import { Progress } from "@/components/ui/progress"
import {
  Stepper,
  StepperItem,
  StepperSeparator,
  StepperTrigger,
  StepperContent,
} from "@/components/ui/stepper"

const steps = [
  { id: "step1", title: "Initialize", duration: 2000 },
  { id: "step2", title: "Processing", duration: 3000 },
  { id: "step3", title: "Finalizing", duration: 1500 },
  { id: "step4", title: "Complete", duration: 0 },
]

export function ProgressStepper() {
  const [currentStep, setCurrentStep] = useState("step1")
  const [progress, setProgress] = useState(0)
  const [isProcessing, setIsProcessing] = useState(false)

  const currentIndex = steps.findIndex(step => step.id === currentStep)
  const currentStepData = steps[currentIndex]

  const startProcess = () => {
    setIsProcessing(true)
    setProgress(0)
    setCurrentStep("step1")
    processSteps(0)
  }

  const processSteps = (stepIndex: number) => {
    if (stepIndex >= steps.length) {
      setIsProcessing(false)
      return
    }

    const step = steps[stepIndex]
    setCurrentStep(step.id)

    if (step.duration > 0) {
      let currentProgress = 0
      const interval = setInterval(() => {
        currentProgress += 2
        setProgress(currentProgress)
        
        if (currentProgress >= 100) {
          clearInterval(interval)
          setTimeout(() => {
            processSteps(stepIndex + 1)
          }, 200)
        }
      }, step.duration / 50)
    } else {
      setProgress(100)
      setIsProcessing(false)
    }
  }

  return (
    <div className="w-full max-w-2xl mx-auto space-y-8">
      <Stepper value={currentStep} className="w-full">
        <div className="flex items-center justify-between">
          {steps.map((step, index) => (
            <div key={step.id} className="flex items-center">
              <StepperItem value={step.id}>
                <StepperTrigger 
                  className={`w-10 h-10 ${
                    index < currentIndex ? 'bg-green-500 text-white' :
                    index === currentIndex ? 'bg-primary text-primary-foreground' :
                    'bg-muted text-muted-foreground'
                  }`}
                >
                  {index < currentIndex ? '✓' : index + 1}
                </StepperTrigger>
              </StepperItem>
              {index < steps.length - 1 && (
                <StepperSeparator 
                  className={`flex-1 mx-4 ${
                    index < currentIndex ? 'bg-green-500' : 'bg-muted'
                  }`} 
                />
              )}
            </div>
          ))}
        </div>
        
        <div className="mt-8">
          {steps.map((step) => (
            <StepperContent key={step.id} value={step.id}>
              <div className="text-center space-y-4">
                <h3 className="text-lg font-semibold">{step.title}</h3>
                {step.duration > 0 && isProcessing && currentStep === step.id && (
                  <div className="space-y-2">
                    <Progress value={progress} className="w-full" />
                    <p className="text-sm text-muted-foreground">
                      {progress.toFixed(0)}% complete
                    </p>
                  </div>
                )}
                {step.id === "step4" && (
                  <p className="text-green-600 font-medium">Process completed successfully!</p>
                )}
              </div>
            </StepperContent>
          ))}
        </div>
      </Stepper>
      
      <div className="text-center">
        <Button 
          onClick={startProcess} 
          disabled={isProcessing}
          className="w-full"
        >
          {isProcessing ? 'Processing...' : 'Start Process'}
        </Button>
      </div>
    </div>
  )
}
```

## API Reference

### Stepper

The root stepper component.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | The controlled active step |
| `defaultValue` | `string` | - | The default active step (uncontrolled) |
| `onValueChange` | `(value: string) => void` | - | Callback when active step changes |
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Layout orientation |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The stepper content |

### StepperItem

A single step in the stepper.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | The step identifier (required) |
| `disabled` | `boolean` | `false` | Whether the step is disabled |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The step content |

### StepperTrigger

The clickable trigger for a step.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The trigger content |
| `asChild` | `boolean` | `false` | Render as child component |

**Default Styles:**
- `inline-flex items-center justify-center` - Layout
- `rounded-full border-2 border-muted` - Appearance
- `bg-background text-muted-foreground` - Colors
- `w-8 h-8 text-sm font-medium` - Size and typography
- `transition-colors` - Smooth transitions
- `hover:bg-accent hover:text-accent-foreground` - Hover states
- `focus-visible:outline-none focus-visible:ring-2` - Focus states
- `disabled:pointer-events-none disabled:opacity-50` - Disabled state
- `data-[state=active]:border-primary data-[state=active]:bg-primary` - Active state
- `data-[state=active]:text-primary-foreground` - Active text color

### StepperContent

The content area for a step.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | - | The step identifier (required) |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | The content |

**Default Styles:**
- `mt-2 ring-offset-background` - Spacing and focus ring offset
- `focus-visible:outline-none focus-visible:ring-2` - Focus states
- `focus-visible:ring-ring focus-visible:ring-offset-2` - Focus ring styling

### StepperSeparator

A visual separator between steps.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Separator orientation |
| `className` | `string` | - | Additional CSS classes |

**Default Styles:**

**Horizontal:**
- `flex-1 h-[2px] bg-muted` - Layout and appearance

**Vertical:**
- `w-[2px] h-4 bg-muted` - Layout and appearance

## Features

- **Flexible Layout**: Support for horizontal and vertical orientations
- **Controlled/Uncontrolled**: Support for both usage patterns
- **Accessible**: Full keyboard navigation and screen reader support
- **Customizable**: Easy to style and extend
- **Step States**: Support for active, completed, and disabled states
- **Progress Tracking**: Visual indication of progress
- **Form Integration**: Perfect for multi-step forms
- **Responsive**: Adapts to different screen sizes

## Accessibility

- Uses proper ARIA attributes (`role="tablist"`, `aria-selected`, etc.)
- Keyboard navigation with arrow keys and Tab
- Focus management between steps
- Screen reader announcements
- Proper labeling and descriptions
- Support for disabled states

## Best Practices

1. **Clear Labels**: Use descriptive step titles and descriptions
2. **Logical Flow**: Ensure steps follow a logical sequence
3. **Progress Indication**: Show clear progress through the process
4. **Validation**: Validate each step before allowing progression
5. **Error Handling**: Provide clear error feedback
6. **Save Progress**: Consider saving progress for long forms
7. **Mobile Optimization**: Ensure good mobile experience
8. **Accessibility**: Test with keyboard and screen readers
9. **Loading States**: Handle loading states appropriately
10. **Consistent Styling**: Maintain consistent appearance

## Troubleshooting

### Steps Not Changing

Ensure proper value management:

```tsx
const [currentStep, setCurrentStep] = useState("step1")

<Stepper value={currentStep} onValueChange={setCurrentStep}>
  {/* steps */}
</Stepper>
```

### Styling Issues

Ensure Tailwind CSS is properly configured and component styles are included in your build.

### Accessibility Issues

Ensure proper labeling:

```tsx
<StepperTrigger aria-label="Step 1: Personal Information">
  1
</StepperTrigger>
```

### Performance Issues

For complex step content, consider lazy loading:

```tsx
<StepperContent value="step1">
  {currentStep === "step1" && <ComplexStepContent />}
</StepperContent>
```

### Mobile Responsiveness

For mobile, consider using vertical orientation or simplified layout:

```tsx
<Stepper 
  orientation={isMobile ? "vertical" : "horizontal"}
  className={isMobile ? "space-y-4" : ""}
>
  {/* steps */}
</Stepper>
```
