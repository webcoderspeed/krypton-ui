# Tabs

A set of layered sections of content—known as tab panels—that are displayed one at a time. Perfect for organizing related content and improving user experience by reducing cognitive load.

## Installation

The tabs component is built on top of Radix UI's Tabs primitive.

```bash
npm install @radix-ui/react-tabs
```

## Import

```tsx
import {
  Tabs,
  TabsContent,
  TabsList,
  TabsTrigger,
} from "@/components/ui/tabs"
```

### Controlled Tabs

```tsx
import { useState } from "react"
import {
  Tabs,
  TabsContent,
  TabsList,
  TabsTrigger,
} from "@/components/ui/tabs"
import { Button } from "@/components/ui/button"

export function ControlledTabs() {
  const [activeTab, setActiveTab] = useState("overview")

  const tabs = [
    { value: "overview", label: "Overview" },
    { value: "analytics", label: "Analytics" },
    { value: "reports", label: "Reports" },
    { value: "notifications", label: "Notifications" },
  ]

  return (
    <div className="w-full max-w-2xl">
      <div className="mb-4 flex space-x-2">
        {tabs.map((tab) => (
          <Button
            key={tab.value}
            variant={activeTab === tab.value ? "default" : "outline"}
            size="sm"
            onClick={() => setActiveTab(tab.value)}
          >
            {tab.label}
          </Button>
        ))}
      </div>
      
      <Tabs value={activeTab} onValueChange={setActiveTab}>
        <TabsList className="hidden">
          {tabs.map((tab) => (
            <TabsTrigger key={tab.value} value={tab.value}>
              {tab.label}
            </TabsTrigger>
          ))}
        </TabsList>
        
        <TabsContent value="overview" className="mt-0">
          <div className="rounded-lg border p-6">
            <h3 className="text-lg font-semibold">Overview</h3>
            <p className="mt-2 text-muted-foreground">
              Get a high-level view of your dashboard metrics and key performance indicators.
            </p>
          </div>
        </TabsContent>
        
        <TabsContent value="analytics" className="mt-0">
          <div className="rounded-lg border p-6">
            <h3 className="text-lg font-semibold">Analytics</h3>
            <p className="mt-2 text-muted-foreground">
              Dive deep into your data with detailed analytics and insights.
            </p>
          </div>
        </TabsContent>
        
        <TabsContent value="reports" className="mt-0">
          <div className="rounded-lg border p-6">
            <h3 className="text-lg font-semibold">Reports</h3>
            <p className="mt-2 text-muted-foreground">
              Generate and view comprehensive reports for your business.
            </p>
          </div>
        </TabsContent>
        
        <TabsContent value="notifications" className="mt-0">
          <div className="rounded-lg border p-6">
            <h3 className="text-lg font-semibold">Notifications</h3>
            <p className="mt-2 text-muted-foreground">
              Manage your notification preferences and settings.
            </p>
          </div>
        </TabsContent>
      </Tabs>
    </div>
  )
}
```

## API Reference

### Tabs

The root tabs container.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `defaultValue` | `string` | - | The default active tab |
| `value` | `string` | - | The controlled active tab |

## Troubleshooting

### Tabs Not Switching

Ensure `value` props match between triggers and content:

```tsx
<TabsTrigger value="tab1">Tab 1</TabsTrigger>
<TabsContent value="tab1">Content 1</TabsContent>
```

### Styling Issues

Ensure Tailwind CSS is properly configured and component styles are included.

### Accessibility Issues

Verify proper ARIA attributes and keyboard navigation:

```tsx
<Tabs>
  <TabsList aria-label="Settings tabs">
    <TabsTrigger value="general">General</TabsTrigger>
  </TabsList>
  <TabsContent value="general">
    Content
  </TabsContent>
</Tabs>
```

### Content Not Showing

Check that `defaultValue` or `value` matches one of your tab values:

```tsx
<Tabs defaultValue="existing-tab-value">
  <TabsTrigger value="existing-tab-value">Tab</TabsTrigger>
  <TabsContent value="existing-tab-value">Content</TabsContent>
</Tabs>
```

### Performance Issues

For heavy content, consider lazy loading:

```tsx
<TabsContent value="heavy-tab">
  {activeTab === "heavy-tab" && <HeavyComponent />}
</TabsContent>
```
