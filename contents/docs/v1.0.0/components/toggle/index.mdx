## Real-World Examples

### Text Editor Toolbar

A comprehensive text editor toolbar with formatting options using toggle buttons.

<CodePreview 
  code={`import { useState } from "react";
import { Toggle } from "@/components/ui/toggle";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Separator } from "@/components/ui/separator";
import { 
  Bold, 
  Italic, 
  Underline, 
  AlignLeft, 
  AlignCenter, 
  AlignRight,
  List,
  ListOrdered
} from "lucide-react";

export default function ToggleTextEditor() {
  const [formatting, setFormatting] = useState({
    bold: false,
    italic: false,
    underline: false,
    alignLeft: true,
    alignCenter: false,
    alignRight: false,
    bulletList: false,
    numberedList: false
  });

  const handleToggle = (key: string) => {
    setFormatting(prev => {
      const newState = { ...prev, [key]: !prev[key] };
      
      // Handle alignment exclusivity
      if (key.startsWith('align')) {
        newState.alignLeft = key === 'alignLeft';
        newState.alignCenter = key === 'alignCenter';
        newState.alignRight = key === 'alignRight';
      }
      
      // Handle list exclusivity
      if (key.includes('List')) {
        if (key === 'bulletList') {
          newState.numberedList = false;
        } else if (key === 'numberedList') {
          newState.bulletList = false;
        }
      }
      
      return newState;
    });
  };

  const getFormattingPreview = () => {
    const styles = [];
    if (formatting.bold) styles.push('font-bold');
    if (formatting.italic) styles.push('italic');
    if (formatting.underline) styles.push('underline');
    
    let alignment = 'text-left';
    if (formatting.alignCenter) alignment = 'text-center';
    if (formatting.alignRight) alignment = 'text-right';
    
    return { styles: styles.join(' '), alignment };
  };

  const { styles, alignment } = getFormattingPreview();

  return (
    <Card className="w-full max-w-2xl">
      <CardHeader>
        <CardTitle>Text Editor</CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        {/* Formatting Toolbar */}
        <div className="flex flex-wrap items-center gap-1 p-2 border rounded-lg bg-muted/50">
          {/* Text Formatting */}
          <div className="flex items-center gap-1">
            <Toggle
              pressed={formatting.bold}
              onPressedChange={() => handleToggle('bold')}
              aria-label="Toggle bold"
              size="sm"
            >
              <Bold className="h-4 w-4" />
            </Toggle>
            <Toggle
              pressed={formatting.italic}
              onPressedChange={() => handleToggle('italic')}
              aria-label="Toggle italic"
              size="sm"
            >
              <Italic className="h-4 w-4" />
            </Toggle>
            <Toggle
              pressed={formatting.underline}
              onPressedChange={() => handleToggle('underline')}
              aria-label="Toggle underline"
              size="sm"
            >
              <Underline className="h-4 w-4" />
            </Toggle>
          </div>

          <Separator orientation="vertical" className="h-6" />

          {/* Alignment */}
          <div className="flex items-center gap-1">
            <Toggle
              pressed={formatting.alignLeft}
              onPressedChange={() => handleToggle('alignLeft')}
              aria-label="Align left"
              size="sm"
            >
              <AlignLeft className="h-4 w-4" />
            </Toggle>
            <Toggle
              pressed={formatting.alignCenter}
              onPressedChange={() => handleToggle('alignCenter')}
              aria-label="Align center"
              size="sm"
            >
              <AlignCenter className="h-4 w-4" />
            </Toggle>
            <Toggle
              pressed={formatting.alignRight}
              onPressedChange={() => handleToggle('alignRight')}
              aria-label="Align right"
              size="sm"
            >
              <AlignRight className="h-4 w-4" />
            </Toggle>
          </div>

          <Separator orientation="vertical" className="h-6" />

          {/* Lists */}
          <div className="flex items-center gap-1">
            <Toggle
              pressed={formatting.bulletList}
              onPressedChange={() => handleToggle('bulletList')}
              aria-label="Bullet list"
              size="sm"
            >
              <List className="h-4 w-4" />
            </Toggle>
            <Toggle
              pressed={formatting.numberedList}
              onPressedChange={() => handleToggle('numberedList')}
              aria-label="Numbered list"
              size="sm"
            >
              <ListOrdered className="h-4 w-4" />
            </Toggle>
          </div>
        </div>

        {/* Preview */}
        <div className="p-4 border rounded-lg bg-background min-h-[100px]">
          <p className={\`\${styles} \${alignment}\`}>
            This is a preview of your formatted text. The formatting will be applied based on your toolbar selections.
          </p>
          {formatting.bulletList && (
            <ul className="list-disc list-inside mt-2">
              <li>Bullet point one</li>
              <li>Bullet point two</li>
            </ul>
          )}
          {formatting.numberedList && (
            <ol className="list-decimal list-inside mt-2">
              <li>Numbered item one</li>
              <li>Numbered item two</li>
            </ol>
          )}
        </div>
      </CardContent>
    </Card>
  );
}`} 
  language="tsx" 
  preview={<ToggleTextEditor />} 
/>

## API Reference

### Toggle

The main toggle component built on top of Radix UI Toggle primitive.

#### Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `variant` | `"default" \| "outline"` | `"default"` | The visual style variant |
| `size` | `"default" \| "sm" \| "lg"` | `"default"` | The size of the toggle |
| `pressed` | `boolean` | `undefined` | The controlled pressed state |
| `defaultPressed` | `boolean` | `false` | The default pressed state when uncontrolled |
| `onPressedChange` | `(pressed: boolean) => void` | `undefined` | Callback fired when the pressed state changes |
| `disabled` | `boolean` | `false` | Whether the toggle is disabled |
| `className` | `string` | `undefined` | Additional CSS classes |
| `children` | `ReactNode` | `undefined` | The content to display inside the toggle |

#### Data Attributes

| Attribute | Values | Description |
| --- | --- | --- |
| `[data-state]` | `"on" \| "off"` | The current pressed state |
| `[data-disabled]` | Present when disabled | Indicates the toggle is disabled |

## Keyboard Navigation

| Key | Action |
| --- | --- |
| **Space** | Toggles the pressed state |
| **Enter** | Toggles the pressed state |
| **Tab** | Moves focus to the next focusable element |
| **Shift + Tab** | Moves focus to the previous focusable element |

## Accessibility

The Toggle component follows accessibility best practices:

- Uses semantic button element with proper ARIA attributes
- Supports keyboard navigation with Space and Enter keys
- Provides clear focus indicators
- Includes proper state announcements for screen readers
- Supports disabled state with appropriate ARIA attributes
- Uses `aria-pressed` to indicate the current state
- Includes `aria-label` or `aria-labelledby` for proper labeling

## Best Practices

1. **Provide clear labels**: Always include descriptive `aria-label` or visible text
2. **Use appropriate variants**: Choose outline variant for secondary actions
3. **Consider state persistence**: Implement proper state management for user preferences
4. **Group related toggles**: Use consistent spacing and alignment for toggle groups
5. **Provide feedback**: Consider visual or haptic feedback for state changes
6. **Test accessibility**: Ensure proper keyboard navigation and screen reader support
